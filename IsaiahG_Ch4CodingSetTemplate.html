# remember to include comments like this one for most of your code

# Setting the days
days = 1

# Setting the total amount
total = 0

# Making the loop for bugs collected
while days < 6:
    bugs = float(input(f"please enter the number of bugs collected on day {days}"))
    total += bugs
    days += 1

# Display the total
print (f"The total amount of bugs collected is {total}")
