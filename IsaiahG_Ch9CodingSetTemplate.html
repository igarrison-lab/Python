def main(user_number):
    # creating the rooms dictionary
    rooms_dict = {"CS101":3004, "CS102":4501, "CS103":6755, "NT110":1244, "CM241":1411}
    # creating the instructors dictionary
    instructors_dict = {"CS101":"Haynes", "CS102":"Alvarado", "CS103":"Rich", "NT110":"Burke", "CM241":"Lee"}
    # creating the meeting times dictionary
    meetingtimes_dict = {"CS101":"8:00 a.m.", "CS102":"9:00 a.m.", "CS103":"10:00 a.m.", "NT110":"11:00 a.m.", "CM241":"12:00 p.m."}
    # displaying the attached rooms 
    user_room = rooms_dict.get(user_number)
    print(f"Your attached room number is {user_room}")
    # displaying the attached instructors 
    user_instuctor = instructors_dict.get(user_number)
    print(f"Your attached instructor is {user_instuctor}")
    # displaying the attached meeting times 
    user_meettimes = meetingtimes_dict.get(user_number)
    print(f"Your attached meeting time is {user_meettimes}")

# running the function
if __name__ == "__main__":
    # asking user for course number
